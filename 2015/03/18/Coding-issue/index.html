LL f(LL a, LL b)
{
    if(a<b) return f(b, a);
    LL ans=0;
    while(a>1 && b>1 && a!=b)
    {
        ans+=(a/b);
        a%=b;
        if(a<b) swap(a, b);
    }
    if(a==b) return ans+1;
    else if(a==1) return ans+b;
    else if(b==1) return ans+a;
    return ans;
}
